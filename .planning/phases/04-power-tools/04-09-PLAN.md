**useResourceMonitor.ts** — composable that polls bollard stats every 2s:

```typescript
import { ref, onMounted, onUnmounted } from 'vue'
import { typedError } from '@/ipc/bindings'

export interface ResourcePoint { time: string; cpu: number; mem: number }

export function useResourceMonitor(projectId: string) {
  const series = ref<ResourcePoint[]>([])
  let intervalId: ReturnType<typeof setInterval> | null = null

  async function poll() {
    const r = await typedError('get_resource_stats', { project_id: projectId }).catch(() => null)
    if (r?.status === 'ok') {
      series.value = [...series.value.slice(-30), {  // keep last 30 points (60 seconds)
        time: new Date().toLocaleTimeString(),
        cpu: r.data.cpu_percent,
        mem: r.data.mem_mb,
      }]
    }
  }

  onMounted(() => { poll(); intervalId = setInterval(poll, 2000) })
  onUnmounted(() => { if (intervalId) clearInterval(intervalId) })

  return { series }
}
```
